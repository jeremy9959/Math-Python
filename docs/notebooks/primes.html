def is_prime(x):
    d = 2
    while d * d <= x:
        if x % d == 0:
            return False
        d += 1
    return True

Euclid’s Algorithm

def qrem(x,a):
    q = x//a 
    if x-q*a <0:
        q=q+1 
    r = x-q*a
    return q,r
def check(a,b):
    q,r = qrem(a,b)
    print(q,r)
    assert r>=0 and r<np.abs(b)
    assert a == q*b+r
def euclid(a,b):
    if a==0 and b==0:
        return 0,0,0
    if a==0:
        return 0,1,b 
    if b==0:
        return 1,0,a 
    a0, a1 = 1,0
    b0, b1 = 0,1
    u0, u1 = a,b
    while u1!=0:
        q, r = qrem(u0,u1) 
        u0,u1 = u1, u0-q*u1 
        a0,a1 = a1, a0-q*a1
        b0, b1 = b1, b0-q*b1 

    assert a0*a+b0*b==u0
    return a0, b0, u0
